import javafx.scene.layout.StackPane;
import javafx.scene.paint.Color;
import javafx.stage.Stage;


public class FxgTest extends Application {
    @Override
    public void start(Stage stage) {
        $className control = new $className();
        // depracted but needed as a workaround for custom components
        // see also http://javafx-jira.kenai.com/browse/RT-16647
        control.impl_processCSS(true);
        control.setPrefSize($width, $height);

        StackPane pane = new StackPane();
        pane.getChildren().add(control);

        Scene scene = new Scene(pane, $width, $height);
        scene.fillProperty().set(Color.DARKGREY);

        stage.setTitle("JavaFX Custom Control");
        stage.setScene(scene);
        stage.show();

        //control.setValue(1.0);
    }

    public static void main(String[] args) {
        Application.launch(args);
    }
